You can manage database resources using Teleport's [Terraform
provider](docks/pages/management/dynamic-resources/terraform-provider.mdx).

Add the following resource to your Terraform deployment and run `terraform
apply`:
```
resource "teleport_database" "{{ matcher }}-database" {
  metadata = {
    name        = "{{ matcher }}-database"
    description = "Example database"

    # Match database resource tags where tag:env=prod.
    # "teleport.dev/origin" = "dynamic" must be specified.
    labels = {
      "teleport.dev/origin" = "dynamic"
      "env"                 = "prod"
    }
  }

  spec = {
    protocol  = "{{ protocol }}"
    uri       = "<Var name="{{ example }}" />"
    aws = {                                
      account_id = "<Var name="aws-account-id"/>"
    }
  }
}
```
